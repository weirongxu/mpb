Class MultiProgressBar

MultiProgressBar

example
const multi = MultiProgressBar.create({
  title: 'download',
});
const bar1 = multi.createBar({
  title: 'progress',
  total: 100,
  clean: true,
});
bar1.start();
const bar2 = multi.startBar({
  title: 'progress',
  total: 100,
  clean: true,
});
bar1.update(10);
bar1.tick(10);
bar2.update(10);
bar2.tick(10);
bar2.forceCompleted();
